```python

# 행(row)으로 되어있는 데이터, 열(column)로 나열

테스트용_data = 테스트용_data.to_numpy().reshape(-1, 1)
# 모델 학습
lr.fit(훈련용_data, 훈련용_target)

# 테스트용_data로 예측
print(테스트용_data)
```
